    #监听自定义事件$emit

    <div id="root">
  <cupon @applied="onCuponValue" v-model="message"><&#47;cupon>
  <p v-if="isShow">{{ message }}<&#47;p>
<&#47;div>
    1
    2
    3
    4
    Vue.component('cupon', {
  template: '&lt;input type="text" @blur="onCuponValue"&gt;',
  methods: {
    onCuponValue() {
      this.$emit('applied', this.$el.value);
    }
  }
})

let app = new Vue({
  el: '#root',

  data: {
    isShow: false,
    message: ''
  },

  methods: {
    onCuponValue(val) {
      if (!val.length) {
        this.isShow = true;
        this.message = '值不能为空！';
      } else {
        this.isShow = false;
      }
    }
  }
})
